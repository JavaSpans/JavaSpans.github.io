package 学习内容;

public class Final关键字 {
	/*
	 * final 关键
	 * public private  --权限修饰符
	 * static --修饰符
	 * void 四种八类 --数据类型
	 * class
	 * this super 
	 * final(最终，最后) 关键字 --修饰符
	 * 1.修饰类
	 * 2.修饰方法
	 * 3.修饰变量
	 *	final 修饰的类不能有子类（不能被继承）
	 *	final 修饰的方法不能被重写,但是可以被继承
	 *	final 修饰的变量初始化后就不能重新赋值
	 *	final static int i = 30；这是常量的定义
	 *
	 */
	public static void main(String[] args) {
		Rectangle c = new Rectangle();
	}
}
